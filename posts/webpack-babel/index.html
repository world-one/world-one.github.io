@babel/core를 이용하여 실행
//runBabel.js
const babel = require('@babel/core');
const ffs = require('fs');

const filename = './src/code.js';
const source = fss.readFileSync(filename, 'utf8'); //컴파일 할 파일
const presets = ['@babel/preset-react'];
const plugins = [
  '@babel/plugin-transform-teamplate-literals',
  '@babel/plugin-transform-arrow-functions',
];
const { code } = babel.transformSync(source, { // transformSync 바벨 실행
  filename,
  presets,
  plugins,
  configFile: false, //babel.config.js 설정 파일을 사용하지 않는다.
});

console.log(code); //파일 저장은 fs 모듈을 이용

이 방법은 자유도가 높다.

바벨의 컴파일 단계

  • 파싱 단계 : 입력된 코드로부터 AST(abstract syntax tree) 생성
  • 변환 단계 : AST를 원하는 형태로 변환
  • 생성 단계 : AST를 코드로 출력
const { ast } = babel.transformSync(sources, { // 코드는 생성하지 않고 AST만 생성, 공통 부분만 추가 
  filename,
  ast: true,
  code: false,
  presets,
  configFile: false,
});

const { code: code1 } = babel.transformSync(ast, source, { // transformSync 바벨 실행
  filename,  
  plugins : ['@babel/plugin-transform-teamplate-literals'],
  configFile: false, //babel.config.js 설정 파일을 사용하지 않는다.
});

const { code: code2 } = babel.transformSync(ast, source, { // transformSync 바벨 실행
  filename,
  plugins: ['@babel/plugin-transform-arrow-functions'],
  configFile: false, //babel.config.js 설정 파일을 사용하지 않는다.
});

설정의 갯수가 많아질수록 효율이 높아지는 방식이라고 한다.

확장성과 유연성
  • extends 다른 설정 파일을 가져와서 확장할 수 있다.
//common/.babelrc
{
  "presets": ["@babel/preset-react"],
  "plugins": [
    [
      "@babel/plugin-transfrom-template-literals",
      {
        "loose": true //concat 대신 + 연산자 사용
      }
    ]
  ]
}
//src/example-extends/.babelrc
{
  "extends": "../../common/.babelrc",
  "plugins": [
    "@babel/plugin-transform-arrow-functions",
    "@babel/plugin-transform-template-literals",
  ]
}

core-js

바벨에서 폴리필을 위해 공식적으로 제공
가장 간단한 방법은 자바스크립트 코드로 불러오는 것이다. 

import 'core-js';

이는 해당 모듈의 모든 폴리필을 포함한다.

웹팩에서 설정할 때는 아래와 같이 entry 속성에 추가해주면 된다.

module.exportss = {
  entry: ['core-js', './src/index.js'],
}

필요 없는 폴리필까지 번들에 포한되어 파일의 크기가 커진다.
만약 파일의 크기가 문제가 된다면 필요한 폴리필만 호출해서 사용한다. 

import 'core-js/features/promise';
import 'core-js/features/object/values';
import 'core-js/features/array/includes';
@babel/preset-env

실행환경에 대한 정보를 설정하면 자동으로 필요한 기능을 주입해준다. 

//babel.config.js
const presets = [
  [
    '@babel/preset-env',
    {
      targets: '> 0.25%, not dead', 
    },
  ],
];

module.exports = { presets };

targets: ’> 0.25%, not dead’, : 시장점유율이 0.25% 이상이고 업데이트가 종료되지 않은 브라우저
browserslist라는 패키지의 문법 

const presets = [
  [
    '@babel/preset-env',
    {
      targets: {
        chrome: '40',
      },
      useBuiltiIns: 'entry', // 폴리필과 관련된 설정, entry는 지원하는 브라우저에서 필요한 폴리필만 포함, 
                            //usage는 코드에서 사용된 기능의 폴리필만 불러옴
      corejs: { version: 3, proposals: true }, //babel에게 corejs의 버전을 알려준다.
    }
  ]
];

usage를 이용하덜라도 불필요한 폴리필을 불러오기도 하는데 이는 변수의 타입을 추론하지 못하기 때문이다.
그래서 보수적으로 폴리필을 추가하게 되는데, 이는 타입스크립트를 사용하면 해결된다.
